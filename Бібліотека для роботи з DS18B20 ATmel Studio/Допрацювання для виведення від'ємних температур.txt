Программа не корректно работает с датчиком, если тот будет измеряет температуру меньше 0 °C. Я немного дописал код, может кому понадобится.

Необходимо вместо:
  sendcharlcd(tt/10+0x30);//Преобразуем число в код числа
  sendcharlcd(tt%10+0x30);//Преобразуем число в код числа
написать следующее:
  if (tt<=0x7D) //Проверка условия что температура положительная
  {
   sendcharlcd((tt)/10+0x30);
   sendcharlcd((tt)%10+0x30);
  } 
  else
  {
   str_lcd("-");
   sendcharlcd((~tt+1)/10-0x50);//Преобразуем число в код числа
   sendcharlcd((~tt+1)%10+0x30);//Преобразуем число в код числа
  }
